747     * update()
748     * Engine has requested this state update itself.
749     * @param gl OpenGL context
750     */
751    @Override
752    public void update(GL10 gl)
753    {
754        // Handle input
755        if (_showMenu) {
756            _input.process(_cam, _toolbelt, _grid, _menuGuiManager, false);
757        } else {
758            _input.process(_cam, _toolbelt, _grid, _guiManager, true);
759        }
760        
761        // Update game objects
762        for (IGameObject object : _objects.values()) {
763            object.update();
764        }
765        
766        // collision detection
767        LightBeamCollection beams = _lightPath.getLightPaths();
768        for (int i = 0; i < beams.size(); i++) 
769        {
770            LightBeam lightBeam = beams.get(i);
771            int savedBeamSize = lightBeam.size();
772 
773            for (int j = 0; j < lightBeam.size(); j++)        
774            {
775                IGameObject minObj = null;
776                float minDist = Light.LIGHT_INFINITY;
777                float curDist = 0.0f;
778                
779                Light l = lightBeam.get(j);
780                
781                // if light has an end point adjust it for max distance
782                if (l.getEndTouchedObject() != null)
783                {
784                    minDist = (float)Colliders.distance(l.getEndTouchedObject().getPosition(), l.getPosition());
785                }
786                
787                // loop through all objects to find min dist collision
788                for (IGameObject o : _objects.values())
789                {
790                    // ignore collisions with objects we know light
791                    // such a HACK
792                    if (o == l.getStartTouchedObject())
793                    {
794                        continue;
795                    }
796                    
797                    if (o == l.getEndTouchedObject())
798                    {
799                        //minObj = null;
800                        continue;
801                    }
802                    
803                    // get collision point, compare to find min
804                    Vector3f colPoint = Colliders.collide(o.getCollisionSphere(), l.getRay());
805                    if (colPoint != null)
806                    {
807                        curDist = (float)Colliders.distance(colPoint, l.getPosition());
808                        if (curDist <= minDist)
809                        {
810                            minDist = curDist;
811                            minObj = o;                            
812                        }                
813                    }
814                }
815                
816                // we found a min collision
817                if (minObj != null)
818                {
819                    minObj.beamInteract(_lightPath.getLightPaths(), i, j);
820                }
821                
822                // check if the beam was modified and we are done
823                if (lightBeam.size() < savedBeamSize)
824                {
825                    break;
826                }
827                
828            }
829            
830        }
831        
832        // level end check
833        if (!_complete) {
834            boolean bLevelComplete = true;
835            for (Receptor r : _goalObjects)
836            {
837                // all goals must be activated
838                if (!r.getActivated())
839                {
840                    bLevelComplete = false;
841                    break;
842                }
843            }
844 
845            if (bLevelComplete)
846            {
847                logger.info("/--LEVEL COMPLETED--\\");
848                _complete = true;
849                _levelComplete();
850            }
851        }
852        
853        Light.incrementLightScroll();
854    }

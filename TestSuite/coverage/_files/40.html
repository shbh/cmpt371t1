505        _logger.debug("WorldCoords FarPointWW: " + outPointFar[0] + ", " + outPointFar[1] + ", " + outPointFar[2] + ", " + outPointFar[3]);        
506 
507        // Avoid /0 error.
508        if (outPointFar[3] == 0.0) {
509            _logger.error("World coords: Could not calculate world coordinates");
510            return null;
511        }        
512 
513        float xNear = outPointNear[0] / outPointNear[3];
514        float yNear = outPointNear[1] / outPointNear[3];
515        float zNear = outPointNear[2] / outPointNear[3];
516        
517        float xFar = outPointFar[0] / outPointFar[3];
518        float yFar = outPointFar[1] / outPointFar[3];
519        float zFar = outPointFar[2] / outPointFar[3];        
520 
521        Vector3f dir = new Vector3f(xFar - xNear, yFar - yNear, zFar - zNear);
522        dir.normalize();        
523        
524        _logger.debug("World Coords NearPoint: " + xNear + ", " + yNear + ", " + zNear + ")");
525        _logger.debug("World Coords FarPoint: " + xFar + ", " + yFar + ", " + zFar + ")");
526        _logger.debug("World Coords Dir: " + dir.x + ", " + dir.y + ", " + dir.z + ")");
527        
528        _tmpRay.setPosition(xNear, yNear, zNear);
529        _tmpRay.setDirection(dir.x, dir.y, dir.z);
530        
531        return _tmpRay;
532    }

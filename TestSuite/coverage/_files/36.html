445        // the view quat

447        // TODO: might have broken this during port
448        _qConjugate.conjugate(_qRotation);
449 
450        _qRotation.mul(_qView);
451        _qRotation.mul(_qConjugate);
452 
453        _qNewView = _qRotation;
454        // qNewView = ((qRotation * qView) * qRotation.conjugate());
455 
456        // Update the view information by adding the position to the resulting
457        // quaternion.
458        _target.x = _eye.x + _qNewView.x;

531        float screenH = Engine.getInstance().getViewHeight();
532 
533        // touch.Print("World Coords", "Screen touch");
534        //_logger.debug("World Coords Screen: " + screenW + ", " + screenH);
535 
536        // Auxiliary matrix and vectors to deal with ogl.
537        float[] invertedMatrix, transformMatrix, normalizedInPoint, outPointNear, outPointFar;
538        invertedMatrix = new float[16];
539        transformMatrix = new float[16];
540        normalizedInPoint = new float[4];
541        outPointNear = new float[4];
542        outPointFar = new float[4];
543 
544        // Invert y coordinate, as android uses top-left, and ogl bottom-left.
545        int oglTouchY = (int) (screenH - touch.y);
546 
547        // Transform the screen point to clip space in ogl (-1,1) 
548        float winz = 0.0f;
549        normalizedInPoint[0] = (float) ((touch.x) * 2.0f / screenW - 1.0);
550        normalizedInPoint[1] = (float) ((oglTouchY) * 2.0f / screenH - 1.0);
551        normalizedInPoint[2] = 2f * winz - 1.0f;
552        normalizedInPoint[3] = 1.0f;
553                
554        
555        // Obtain the transform matrix and then the inverse. 
556        Matrix.multiplyMM(transformMatrix, 0, _lastProjectionMat, 0,
557                          _lastModelViewMat, 0);
558        Matrix.invertM(invertedMatrix, 0, transformMatrix, 0);
559 
560        // Apply the inverse to the point in clip space, near plane 
561        Matrix.multiplyMV(outPointNear, 0, invertedMatrix, 0, normalizedInPoint, 0);
562        //_logger.debug("WorldCoords NearPointWW: " + outPointNear[0] + ", " + outPointNear[1] + ", " + outPointNear[2] + ", " + outPointNear[3]);
563 
564        // Avoid /0 error.
565        if (outPointNear[3] == 0.0) {
566            _logger.error("World coords: Could not calculate world coordinates");
567            return null;
568        }        
569        
570        // Far plane
571        winz = 1.0f;
572        normalizedInPoint[2] = 2f * winz - 1.0f;
573        Matrix.multiplyMV(outPointFar, 0, invertedMatrix, 0, normalizedInPoint, 0);
574        //_logger.debug("WorldCoords FarPointWW: " + outPointFar[0] + ", " + outPointFar[1] + ", " + outPointFar[2] + ", " + outPointFar[3]);        
575 
576        // Avoid /0 error.
577        if (outPointFar[3] == 0.0) {
578            _logger.error("World coords: Could not calculate world coordinates");
579            return null;
580        }        
581 
582        float xNear = outPointNear[0] / outPointNear[3];
583        float yNear = outPointNear[1] / outPointNear[3];
584        float zNear = outPointNear[2] / outPointNear[3];
585        
586        float xFar = outPointFar[0] / outPointFar[3];
587        float yFar = outPointFar[1] / outPointFar[3];
588        float zFar = outPointFar[2] / outPointFar[3];        
589 
590        Vector3f dir = new Vector3f(xFar - xNear, yFar - yNear, zFar - zNear);
591        dir.normalize();        
592        
593        //_logger.debug("World Coords NearPoint: " + xNear + ", " + yNear + ", " + zNear + ")");
594        //_logger.debug("World Coords FarPoint: " + xFar + ", " + yFar + ", " + zFar + ")");
595        //_logger.debug("World Coords Dir: " + dir.x + ", " + dir.y + ", " + dir.z + ")");
596        
597        _tmpRay.setPosition(xNear, yNear, zNear);
598        _tmpRay.setDirection(dir.x, dir.y, dir.z);
599        
600        return _tmpRay;
601    }
